\begin{frame}{Об'єкт, предмет, методи та мета дослідження}
    \begin{itemize}
        \item \textbf{Об'єкт дослідження.}
            Інтерактивна багатоваріантна колоризація зображень.

        \item \textbf{Предмет дослідження.}
            Моделі глибокого навчання для колоризації зображень,
            а також метрики для їхнього об'єктивного порівняльного аналізу.

        \item \textbf{Методи дослідження.}
            Методи глибокого навчання та комп'ютерного зору,
            методи об'єктивного та суб'єктивного порівняння.

        \item \textbf{Мета дослідження.}
            Розробити модель колоризації на базі запропонованої гібридної архітектури U-Net + BiDir-Mamba,
            створити модульний програмний комплекс інтерактивної колоризації з можливістю вибору різних моделей
            та провести порівняльний аналіз із сучасними рішеннями.
    \end{itemize}
\end{frame}
